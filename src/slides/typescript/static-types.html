Statische Typen
  • Typen nur zur Entwicklungszeit vorhanden
  • Werden von Compiler gelöscht
  • Vorteile: Refactorings, Code Completion und Navigation
  • Basis Type: Boolean, Number, String, Array, Enum, Any, Void
  • Eigene Typen über Klassen und Interfaces
function greeter(person: string) {
    return "Hello, " + person;
}

var user = [0, 1, 2]; // user: array

document.body.innerHTML = greeter(user);
